%% Bachelor Thesis --------------------------------------------------------------

% required parameters:
  % \@titlepagefont
  % @hgbisproposal (boolean)
  % \@programtype
  % \@programname
  % \@placeofstudy
  % @dateofsubmission

\newcommand{\@makebachelortitlepage}{%
\begin{titlepage}%
	\begin{german}
		\rule{0mm}{1mm}%
		\vspace*{0mm}%
		\begin{center}%
			\@titlepagefont%
			\setlength{\unitlength}{1cm}%
			\parbox[c][5cm][c]{12cm}{%
				\centering\LARGE\bfseries%
				\setlength{\baselineskip}{25pt}%
				\@title%
			}
			\vskip 15mm%
			{\large\@author}%
			\vskip 15mm%
			\makelogo%
			\vskip 15mm%
			%
      {\Large \@addspaces{BACHELORARBEIT}%
        \ifthenelse{\boolean{@hgbisproposal}}{\space\space(\@addspaces{EXPOS{\`E}})}{}%
      }
			\vskip 4mm%
			%
			{\large eingereicht am} \vskip 1mm {\@programtype}%
			\vskip 4mm%
			{\large\@programname}%
			\vskip 4mm%
			{\large der \@institution\ in \@placeofstudy}%
			\vskip 16mm%
      {\large im \DTMgermanmonthname{\DTMfetchmonth{@dateofsubmission}} \DTMfetchyear{@dateofsubmission}}%
		\end{center}%
		\vfil%
	\end{german}
\end{titlepage}%
}
